What is the difference between function scoping and block scoping. Write a code for differentiating them.

The difference between function scoping and block scoping are as follows:

  • Function scoping uses the var keyword to declare variables, which are scoped to the function in which they are declared.
  • Block scoping uses the let and const keywords to declare variables, which are scoped to the block in which they are declared. 
        function functionScopeExample() {
          if (true) {
            var functionScopedVar = "I am function scope!";
          }
          console.log(functionScopedVar); // This will log: "I am function scoped!"
        }
        
        function blockScopeExample() {
          if (true) {
            let blockScoped = "I am block scope!";
            console.log(blockScoped);
          }
          // console.log(blockScopedVar); // Uncommenting this line will cause an error: ReferenceError
        }
        
        functionScopeExample();
        blockScopeExample();
